Core Set-up

Welcome to the land of large configuration arrays. This page is a quick reference for when you need to write a new module.

module.info

;Comments are prefixed with a semi-colon
;If you want a new line in a value, then wrap with double-quotes.
;Clear cache after changing this file.

name = Your Module's Friendly Name
description = Your modules's short description.
core = 7.x
;package is used to group modules together on the admin screen.
package = My Application

;files are for things Drupal needs to inspect. It is not necessary to include CSS, JS, or *.tpl.php files.
files[] = module.module
files[] = module.install

dependencies[] = machineName

module.install

<?php
/**
* @file
* Install, update, and uninstall functions for your module.
*/

/**
* Implements hook_schema().
*/

function module_schema() {
	//Definition for a database table so that it can be created (or deleted).
	$schema['tableName'] = array(
		'description' => 'A brief description of your table',
		'fields' => array(
			'id' => array(
				'description' => 'primary key',
				'type' => 'int',
				'size' => 'normal', //normal is the default
				'unsigned' => TRUE,
				'not null' => TRUE,
				'default' => 0
			),
			'columnName' => array(
				'description' => 'short description for this column',
				'type' => 'varchar',
				'length' => 2000,
				'not null' => TRUE,
				'default' => ''
			),
		),
		'primary key' => array('id'),

		//In Drupal 7, foreign keys are documentation only; no physical relationship will be created.
		'foreign keys' => array(
			'fkThisTable_fkTable' => array(
				'table' => 'fkTable',
				'columns' => array('id' => 'id'),
			),
		)
	);
	return $schema;
}

module.module file

Most hooks will reside in your *.module file. Here are some common ones.

<?php
/**
* @file
* Short description.
*/

/**
* Implements hook_help().
*/
function module_help($path, $arg) {
	switch ($path) {
		case 'admin/help#module': {
			$helpMsg = '<h3>' . t('Your Modules Name Goes Here') . '</h3>';
			$helpMsg .= '<p>' . t('Some basic help goes here') . '</p>';
			return $helpMsg;
			break;
		}
	}
}

/**
* Implements hook_permission()
*/ 
function module_permission() {
	$newPermissions = array();
	$newPermissions['yourPermissionMachineName'] = array(
		'title' => t('short description'),
		'description' => t('long description')
	);
	return $newPermissions;
}
